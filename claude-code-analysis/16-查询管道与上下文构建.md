## 1. 模块职责概述

查询管道负责将用户的原始输入转化为发送给 Anthropic API 的结构化请求，这个过程包括：

1. **用户输入处理**：解析 slash 命令、文件提及、图片粘贴
2. **系统提示构建**：组合 CLAUDE.md、环境上下文、工具描述
3. **消息序列化**：将内部 Message 格式转换为 API 格式
4. **上下文管理**：Token 计数、压缩决策、上下文窗口优化

## 2. 用户输入处理（`utils/processUserInput/`）

### 2.1 `processUserInput.ts` — 输入处理主函数

```typescript
// 处理用户输入，返回要发送给 LLM 的消息和元数据
export async function processUserInput(
  prompt: string | ContentBlockParam[],
  source: 'prompt' | 'tool_result',
  context: ProcessUserInputContext,
): Promise<ProcessUserInputResult> {
  
  // ─── 1. Slash 命令检测 ─────────────────────────────────
  if (isSlashCommand(prompt)) {
    const cmdResult = await handleSlashCommand(prompt, context)
    
    if (cmdResult.type === 'local') {
      // 本地命令执行完毕，不需要 LLM
      return { messages: cmdResult.messages, shouldQuery: false }
    }
    
    if (cmdResult.type === 'prompt') {
      // 命令转化为发给 LLM 的提示词
      prompt = cmdResult.promptText
    }
  }
  
  // ─── 2. @ 文件提及处理 ──────────────────────────────────
  const { processedPrompt, attachedFiles } = await expandFileMentions(
    prompt, context.cwd
  )
  
  // ─── 3. 图片附件处理 ────────────────────────────────────
  const { finalPrompt, imageAttachments } = await processImageAttachments(
    processedPrompt, context
  )
  
  // ─── 4. 构建 UserMessage ─────────────────────────────────
  const userMessage = createUserMessage(finalPrompt, {
    attachments: [...attachedFiles, ...imageAttachments],
  })
  
  // ─── 5. 工具白名单过滤 ────────────────────────────────────
  const allowedTools = getCommandAllowedTools(context)
  
  return {
    messages: [userMessage],
    shouldQuery: true,
    allowedTools,
    model: getCommandModel(context),  // 命令可能指定模型
  }
}
```

### 2.2 文件提及展开

```typescript
// "@src/main.ts" 语法处理
async function expandFileMentions(
  text: string,
  cwd: string,
): Promise<{ processedText: string; attachedFiles: FileAttachment[] }> {
  const mentionRegex = /@([\w./\-]+\.\w+)/g
  const attachedFiles: FileAttachment[] = []
  
  const processedText = await replaceAsync(text, mentionRegex, async (match, path) => {
    const fullPath = resolve(cwd, path)
    
    if (await exists(fullPath)) {
      const content = await readFile(fullPath, 'utf-8')
      attachedFiles.push({ path: fullPath, content })
      // 替换为内联标记（告诉 LLM 这是文件内容）
      return `[文件: ${path}]`
    }
    
    return match  // 文件不存在，保留原文本
  })
  
  return { processedText, attachedFiles }
}
```

### 3.2 `buildEffectiveSystemPrompt()` — 系统提示构建

```typescript
export async function buildEffectiveSystemPrompt(
  options: SystemPromptOptions,
): Promise<SystemPrompt> {
  const parts: string[] = []
  
  // 1. 基础系统提示
  const basePrompt = await getSystemPrompt()  // constants/prompts.ts
  parts.push(basePrompt)
  
  // 2. 工具描述（Tool.prompt() 方法）
  const toolDescriptions = await buildToolsPrompt(options.tools, options.agents)
  parts.push(toolDescriptions)
  
  // 3. 环境上下文
  const envContext = await enhanceSystemPromptWithEnvDetails(options.env)
  parts.push(envContext)
  
  // 4. 记忆内容（从 CLAUDE.md 读取）
  const memories = await buildMemoryPrompt(options.cwd)
  if (memories) parts.push(memories)
  
  // 5. 用户自定义
  if (options.customSystemPrompt) {
    parts.push(options.customSystemPrompt)
  }
  
  const combined = parts.filter(Boolean).join('\n\n')
  
  // 6. 追加内容（在末尾添加，不影响 prompt cache）
  if (options.appendSystemPrompt) {
    return asSystemPrompt(combined + '\n\n' + options.appendSystemPrompt)
  }
  
  return asSystemPrompt(combined)
}
```

### 4.2 消息序列化（发送给 API 时）

```typescript
// 将内部 Message[] 转换为 API 期望的 MessageParam[]
function serializeMessages(messages: Message[]): MessageParam[] {
  const result: MessageParam[] = []
  
  for (const msg of messages) {
    switch (msg.type) {
      case 'user':
        result.push({
          role: 'user',
          content: serializeUserContent(msg.message.content),
        })
        break
      case 'assistant':
        result.push({
          role: 'assistant',
          content: serializeAssistantContent(msg.message.content),
        })
        break
      // 系统消息、进度消息等不发给 API
    }
  }
  
  return result
}
```

### 4.3 Thinking 消息规则

```typescript
// API 对 thinking blocks 有严格规则：
// 1. thinking/redacted_thinking block 必须在对应的 max_thinking_length > 0 的请求中
// 2. thinking block 不能是最后一个 block（后面必须有 text 或 tool_use）

function filterMessagesForAPI(
  messages: Message[],
  thinkingEnabled: boolean,
): MessageParam[] {
  if (!thinkingEnabled) {
    // 移除所有 thinking blocks（发送给 API 时会出错）
    return removeThinkingBlocks(messages)
  }

  return ensureThinkingIntegrity(messages)
}

// 消息的 token 估算（不精确，但足够用于压缩触发判断）
export function tokenCountWithEstimation(messages: Message[]): number {
  let total = 0
  
  for (const msg of messages) {
    switch (msg.type) {
      case 'user':
      case 'assistant':
        total += estimateContentTokens(msg.message.content)
        break
      case 'system':
        if (msg.subtype === 'compact_boundary') {
          total += estimateContentTokens(msg.summary?.content)
        }
        break
    }
  }
  
  return total
}

// 基于 API 响应中的 usage 字段获取精确 token 数
export function finalContextTokensFromLastResponse(
  messages: Message[],
): number {
  // 从最后一条带 usage 的 assistant 消息中获取精确 input_tokens
  const lastWithUsage = [...messages].reverse().find(
    m => m.type === 'assistant' && m.message.usage?.input_tokens
  )
  return lastWithUsage?.message?.usage?.input_tokens ?? 0
}
```

// 当 token 接近限制时的警告状态
export function calculateTokenWarningState(
  currentTokens: number,
  model: string,
): TokenWarningState {
  const maxTokens = getModelContextWindow(model)
  const ratio = currentTokens / maxTokens
  
  if (ratio >= TOKEN_BLOCKING_RATIO) {  // 默认 0.95
    return { isAtBlockingLimit: true, isAtWarningLimit: true, ratio }
  }
  
  if (ratio >= TOKEN_WARNING_RATIO) {   // 默认 0.85
    return { isAtBlockingLimit: false, isAtWarningLimit: true, ratio }
  }
  
  return { isAtBlockingLimit: false, isAtWarningLimit: false, ratio }
}

## 6. 工具结果预算（`utils/toolResultBudget.ts`）

```typescript
// 防止大型工具结果撑爆上下文窗口
// 每个工具有 maxResultSizeChars 配置，超过则截断

export async function applyToolResultBudget(
  messages: Message[],
  contentReplacementState: ContentReplacementState,
): Promise<Message[]> {
  return messages.map(msg => {
    if (msg.type !== 'user') return msg
    
    // 检查每个 tool_result block
    const newContent = msg.message.content.map(block => {
      if (block.type !== 'tool_result') return block
      
      const toolDef = findToolByToolUseId(block.tool_use_id)
      if (!toolDef?.maxResultSizeChars) return block
      
      const content = extractTextContent(block.content)
      
      if (content.length > toolDef.maxResultSizeChars) {
        // 截断并标记（避免重复截断）
        contentReplacementState.mark(block.tool_use_id)
        return createTruncatedBlock(block, toolDef.maxResultSizeChars)
      }
      
      return block
    })
    
    return { ...msg, message: { ...msg.message, content: newContent } }
  })
}
```

// 环境上下文（每次查询时动态注入到第一条 user 消息）
export async function buildUserContext(
  cwd: string,
  ideInfo?: IDEInfo,
): Promise<UserContext> {
  return {
    // 工作目录信息
    cwd,
    projectRoot: getProjectRoot(),
    
    // 系统信息
    platform: process.platform,        // 'darwin' | 'win32' | 'linux'
    shell: process.env.SHELL ?? 'bash',
    nodeVersion: process.version,
    
    // 当前时间（用于日期相关任务）
    currentTime: new Date().toISOString(),
    
    // IDE 上下文（如果有）
    ...(ideInfo && {
      openFiles: ideInfo.openFiles,
      cursorFile: ideInfo.cursorFile,
      cursorLine: ideInfo.cursorLine,
      selection: ideInfo.selection,
    }),
    
    // Git 信息
    gitBranch: await getCurrentBranch().catch(() => null),
    gitRemote: await getRemoteUrl().catch(() => null),
  }
}

// 用户上下文以 <user_context> XML 标签注入到第一条 user 消息前
export function prependUserContext(
  messages: MessageParam[],
  userContext: UserContext,
): MessageParam[] {
  const contextXml = buildUserContextXml(userContext)
  
  // 将上下文注入到第一条 user 消息
  return [
    {
      role: 'user',
      content: `<user_context>\n${contextXml}\n</user_context>\n\n${messages[0]?.content}`,
    },
    ...messages.slice(1),
  ]
}

## 9. 设计亮点

### 9.1 提示词缓存最大化
系统提示的组织顺序经过精心设计：最稳定的部分（基础行为规范、工具描述）放在最前面，最动态的部分（时间戳、环境信息）放在最后面，最大化 prompt cache 命中率。

### 9.2 XML 标签上下文注入
用户上下文以 `<user_context>...</user_context>` 包裹注入，让 Claude 能清晰区分环境信息和实际对话内容。

### 9.3 文件提及懒加载
`@src/main.ts` 在处理用户输入时才读取文件，而不是预先读取所有文件，避免不必要的 I/O。

### 9.4 Token 预算分层
工具结果预算（单工具级）→ 微压缩（重复模式）→ 自动压缩（整体历史），形成递进式的 token 管理策略。
