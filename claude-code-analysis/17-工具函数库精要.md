```typescript
// expandPath — 展开路径中的 ~（波浪线）
export function expandPath(path: string): string {
  if (path.startsWith('~/')) {
    return join(homedir(), path.slice(2))
  }
  if (path === '~') {
    return homedir()
  }
  return path
}

// 使用示例：
expandPath('~/projects')  // → '/Users/alice/projects'
expandPath('/absolute')   // → '/absolute'（不变）

// trustedPath — 验证路径是否在可信目录内
export function isPathUnderTrustedDirectory(
  filePath: string,
  trustedDirs: string[],
): boolean {
  const resolved = resolve(filePath)
  return trustedDirs.some(dir => resolved.startsWith(resolve(dir)))
}
```

```typescript
  return err instanceof Error && (err as NodeJS.ErrnoException).code === 'ENOENT'
}

export function errorMessage(err: unknown): string {
  if (err instanceof Error) return err.message
  if (typeof err === 'string') return err
  return String(err)
}

export function toError(err: unknown): Error {
  if (err instanceof Error) return err
  return new Error(errorMessage(err))
}
```

```typescript
// 原子写入（先写临时文件，再重命名，防止写入中断导致文件损坏）
export async function writeTextContent(
  filePath: string,
  content: string,
): Promise<void> {
  const tmpPath = `${filePath}.tmp.${randomBytes(4).toString('hex')}`
  
  try {
    await writeFile(tmpPath, content, 'utf-8')
    await rename(tmpPath, filePath)  // 原子操作
  } catch (err) {
    await unlink(tmpPath).catch(() => {})  // 清理临时文件
    throw err
  }
}

// 文件修改时间检测（防止并发修改）
export async function getFileModificationTime(filePath: string): Promise<number> {
  const stats = await fsStat(filePath)
  return stats.mtimeMs
}

export async function findSimilarFile(
  filePath: string,
  cwd: string,
): Promise<string | null> {
  const basename = path.basename(filePath)
  const files = await glob(`**/${basename}`, { cwd, maxDepth: 5 })
  return files[0] ?? null
}
```

```typescript
// 截断长字符串（保留首尾，中间用 ... 替换）
export function truncate(text: string, maxLength: number): string {
  if (text.length <= maxLength) return text
  const half = Math.floor((maxLength - 3) / 2)
  return text.slice(0, half) + '...' + text.slice(-half)
}

// 格式化文件大小（人类可读）
export function formatFileSize(bytes: number): string {
  if (bytes < 1024) return `${bytes}B`
  if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)}KB`
  return `${(bytes / 1024 / 1024).toFixed(1)}MB`
}

// 格式化 USD 费用
export function formatCost(usd: number): string {
  if (usd < 0.01) return `$${(usd * 100).toFixed(2)}¢`
  return `$${usd.toFixed(4)}`
}

// 格式化相对时间
export function formatRelativeDate(timestamp: number): string {
  const diffMs = Date.now() - timestamp
  const diffMins = Math.floor(diffMs / 60_000)
  const diffHours = Math.floor(diffMins / 60)
  const diffDays = Math.floor(diffHours / 24)
  
  if (diffMins < 60) return `${diffMins} 分钟前`
  if (diffHours < 24) return `${diffHours} 小时前`
  if (diffDays < 7) return `${diffDays} 天前`
  return new Date(timestamp).toLocaleDateString('zh-CN')
}
```

## 7. Shell 执行（`utils/Shell.ts`）

```typescript
// exec — 执行 Shell 命令，带超时和信号支持
export async function exec(
  command: string,
  options?: ExecOptions,
): Promise<ExecResult> {
  const {
    cwd = getCwd(),
    timeout = DEFAULT_TIMEOUT_MS,
    signal,
    env = process.env,
  } = options ?? {}
  
  return new Promise((resolve, reject) => {
    const proc = spawn(getShell(), ['-c', command], {
      cwd,
      env: { ...env, ...SAFE_ENV_VARS },
      timeout,
    })
    
    let stdout = ''
    let stderr = ''
    
    proc.stdout.on('data', (data) => { stdout += data.toString() })
    proc.stderr.on('data', (data) => { stderr += data.toString() })
    
    // AbortController 信号支持
    signal?.addEventListener('abort', () => {
      proc.kill('SIGTERM')
      setTimeout(() => proc.kill('SIGKILL'), 5000)  // 5 秒后强制杀死
    })
    
    proc.on('close', (code) => {
      resolve({ stdout, stderr, exitCode: code ?? 0 })
    })
    
    proc.on('error', reject)
  })
}
```

```typescript
// getCwd — 获取当前工作目录（优先返回 Agent 覆盖的 CWD）
export function getCwd(): string {
  // Agent 可以有自己的 CWD（isolation: 'worktree' 时）
  return getAgentCwdOverride() ?? state.cwd
}

// runWithCwdOverride — 临时改变 CWD 执行回调
export async function runWithCwdOverride<T>(
  newCwd: string,
  fn: () => Promise<T>,
): Promise<T> {
  const prevCwd = getCwd()
  setCwd(newCwd)
  try {
    return await fn()
  } finally {
    setCwd(prevCwd)  // 确保恢复，即使抛出异常
  }
}
```

## 9. Signal 响应式原语（`utils/signal.ts`）

Claude Code 自己实现了简单的 Signal（类 SolidJS）：

```typescript
// 创建一对 getter/setter，getter 是响应式的（订阅时会追踪依赖）
export function createSignal<T>(initialValue: T): [() => T, (value: T) => void] {
  let value = initialValue
  const listeners = new Set<() => void>()
  
  const getter = () => {
    // 在 effect 上下文中，自动追踪此 signal
    if (currentEffect) {
      listeners.add(currentEffect)
    }
    return value
  }
  
  const setter = (newValue: T) => {
    value = newValue
    listeners.forEach(l => l())
  }
  
  return [getter, setter]
}

// 使用示例（bootstrap/state.ts 中）：
export const [getCurrentTurnTokenBudget, setCurrentTurnTokenBudget] =
  createSignal<number | undefined>(undefined)
```

```typescript
export function countLinesChanged(
  oldContent: string,
  newContent: string,
): { added: number; removed: number } {
  const diff = diffLines(oldContent, newContent)
  
  return diff.reduce(
    (acc, part) => ({
      added: acc.added + (part.added ? part.count ?? 0 : 0),
      removed: acc.removed + (part.removed ? part.count ?? 0 : 0),
    }),
    { added: 0, removed: 0 },
  )
}
```

```typescript
// JSON.stringify 的慢操作版本（检测性能瓶颈）
export function jsonStringify(value: unknown): string {
  const start = performance.now()
  const result = JSON.stringify(value)
  const elapsed = performance.now() - start
  
  if (elapsed > SLOW_OPERATION_THRESHOLD_MS) {
    logForDebugging(`Slow JSON.stringify: ${elapsed.toFixed(2)}ms`, {
      valueType: typeof value,
      valueSize: result.length,
    })
  }
  
  return result
}
```

## 14. 关键设计决策

### 14.1 错误层级化
错误从底层（`ShellError`）到高层（权限拒绝消息）逐层转化，每层都提供适合自己层次的错误信息。

### 14.2 原子文件写入
所有文件写入操作使用临时文件 + 重命名策略，确保写入中断不会产生损坏的文件。

### 14.3 CWD 动态覆盖
Agent 使用 `runWithCwdOverride` 临时覆盖工作目录，确保在 try/finally 中正确恢复，即使抛出异常。

### 14.4 性能监控内置
慢操作检测（`slowOperations.ts`）内置在关键路径中，而不是依赖外部 profiler，便于生产环境调试。
